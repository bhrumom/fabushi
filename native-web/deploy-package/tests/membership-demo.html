💻 技术实现

前端会员状态检查

async checkMembershipStatus(token) { const response = await fetch('/api/stripe/membership-status', { method: 'GET', headers: { 'Authorization': `Bearer ${token}`, 'Content-Type': 'application/json' } }); if (!response.ok) { throw new Error('认证失败，请重新登录'); } const data = await response.json(); return data.membership; }

发送功能权限控制

async startSending() { // 1. 检查登录状态 const token = localStorage.getItem('authToken'); if (!token) { alert('请先登录后再使用全球法布施功能！'); return; } // 2. 检查会员状态 const membershipStatus = await this.checkMembershipStatus(token); if (!membershipStatus.isActive) { const upgradeMessage = membershipStatus.type === 'trial' && membershipStatus.isExpired ? '您的免费试用期已结束，请升级为付费会员' : '全球法布施功能仅限会员使用，请先开通会员服务'; alert(`🔒 ${upgradeMessage}`); window.location.href = '/membership.html'; return; } // 3. 开始发送流程 this.logMessage('✅ 会员验证通过，开始发送...'); // ... 发送逻辑 }

UI状态控制

function updateMembershipBadge(membership) { const membershipNotice = document.getElementById('membershipNotice'); const startBtn = document.getElementById('startBtn'); if (membership.isActive) { // 隐藏会员提示，恢复按钮 membershipNotice.style.display = 'none'; startBtn.innerHTML = '🚀 开始传送'; } else { // 显示会员提示，禁用按钮 membershipNotice.style.display = 'block'; startBtn.innerHTML = '💎 需要会员'; } }
